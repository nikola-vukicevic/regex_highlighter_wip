def idiosync_parse_12():
	s        = ucitavanje
	redovi   = ucitavanje.split('\\n') 
	tokeni   = []
	s        = ""
	parse_12 = True

	for r in redovi:
		if r == "\\r" or r == "":
			continue
		
		if parse == True:
			if r.startswith("\`\`"):
				parse = False
				continue
			ucitavanje_tokena(r, tokeni)			
		else:
			if r.startswith("~~"):
				parse = True
				tokeni.append(["", "", ""])
				continue
			tokeni.append([r, "", ""])

	for t in tokeni:
		s = s + t[0] + t[1] + t[2] + "\\n"

return s

def idiosync_parse_12():
	s        = ucitavanje
	redovi   = ucitavanje.split('\\n') 
	tokeni   = []
	s        = ""
	parse_12 = True

	for r in redovi:
		if r == "\\r" or r == "":
			continue
		
		if parse == True:
			if r.startswith("\`\`"):
				parse = False
				continue
			ucitavanje_tokena(r, tokeni)			
		else:
			if r.startswith("~~"):
				parse = True
				tokeni.append(["", "", ""])
				continue
			tokeni.append([r, "", ""])

	for t in tokeni:
		s = s + t[0] + t[1] + t[2] + "\\n"

return s

		}
	}

		return _starost;
	}
